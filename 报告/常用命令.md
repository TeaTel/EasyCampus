# 本地推送

git add . && git commit -m "update" && git push origin main

# 服务器拉取并重建

cd /opt/campus\_platform && git pull && docker compose -f docker-compose.prod.yml up -d --build

cd /opt/campus\_platform

docker compose -f docker-compose.prod.yml ps # 查看状态

docker compose -f docker-compose.prod.yml logs -f # 查看日志

docker compose -f docker-compose.prod.yml restart # 重启

docker compose -f docker-compose.prod.yml down # 停止
